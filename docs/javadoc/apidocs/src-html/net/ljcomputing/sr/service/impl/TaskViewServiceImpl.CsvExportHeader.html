017package net.ljcomputing.sr.service.impl;
018
019import net.ljcomputing.exception.PersistenceException;
020import net.ljcomputing.exception.ServiceException;
021import net.ljcomputing.service.impl.AbstractService;
022import net.ljcomputing.sr.model.TaskViewModel;
023import net.ljcomputing.sr.repository.impl.TaskViewModelRepositoryImpl;
024import net.ljcomputing.sr.service.TaskViewService;
025
026import java.io.IOException;
027import java.sql.Timestamp;
028import java.time.LocalDateTime;
029import java.util.List;
030import java.util.Map;
031
032import org.apache.commons.csv.CSVFormat;
033import org.apache.commons.csv.CSVPrinter;
034import org.apache.commons.csv.QuoteMode;
035
036/**
037 * Task view service implementation.
038 * 
039 * @author James G. Willmore
040 */
041public class TaskViewServiceImpl extends AbstractService<TaskViewModel, TaskViewModelRepositoryImpl>
042    implements TaskViewService {
043  
044  /**
045   * Instantiates a new task view service impl.
046   *
047   * @throws ServiceException the service exception
048   */
049  public TaskViewServiceImpl() throws ServiceException {
050    super();
051  }
052
053  /**
054   * @see net.ljcomputing.service.impl.AbstractService#getColumns()
055   */
056  public String[] getColumns() {
057    return COLUMNS;
058  }
059  
060  /**
061   * @see net.ljcomputing.sr.service.TaskViewService#readByTimes(java.time.LocalDateTime, java.time.LocalDateTime)
062   */
063  public List<TaskViewModel> readByTimes(LocalDateTime startTime, LocalDateTime endTime) throws ServiceException {
064    try {
065      Timestamp startTimestamp = Timestamp.valueOf(startTime);
066      Timestamp endTimestamp = Timestamp.valueOf(endTime);
067
068      return repository.readByTimes(startTimestamp, endTimestamp);
069    } catch (PersistenceException exception) {
070      throw new ServiceException(exception);
071    }
072  }
073
074  /**
075   * @see net.ljcomputing.sr.service.TaskViewService#toCsv(java.lang.Appendable, java.util.List)
076   */
077  public void toCsv(Appendable out, List<TaskViewModel> taskViewModels)
078      throws ServiceException {
079    try {
080      Map<String, TaskViewModel> map;
081      
082      CSVFormat format = CSVFormat.DEFAULT.withQuoteMode(QuoteMode.ALL)
083          .withRecordSeparator(System.getProperty("line.separator"));
084      CSVPrinter printer = new CSVPrinter(out, format);
085
086      printer.printRecord(new CsvExportHeader().getHeader());
087
088      for (TaskViewModel taskItem : taskViewModels) {
089        printer.printRecord(taskItem.toValuesList());
090      }
091
092      printer.close();
093    } catch (IOException exception) {
094      throw new ServiceException(exception);
095    }
096  }
097  
098  class CsvExportHeader {
099    final String[] header = {"WBS Name", "Activity Name", "Duration", "Comments"};
100    
101    public String[] getHeader() {
102      return header;
103    }
104  }
105}
